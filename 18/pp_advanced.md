```sh
with open(input().strip(), encoding='utf-8') as in_file, open('forbidden_words.txt', encoding='utf-8') as forbidden:
    s = in_file.read()
    word = list(forbidden.read().strip().split())
    s_lower = s.lower()
    for w in word:
        s_lower = s_lower.replace(w, '*' * len(w))
    out = ''
    for i in range(len(s_lower)):
        if s_lower[i] != '*':
            out += s[i]
        else:
            out += '*'
    print(out)
```
